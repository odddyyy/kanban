POST    /tasks/add      --> adding new todo to the todo-list

#REQUEST HEADER
{
  "Content-Type": "application/json"
}

#REQ.BODY
{
    title: "Learning session on API",
    description: "learning new topic",
}

*authentication

#VALIDATION
title and description cannot be empty

#RESPONSE
{
    "Task created"
}
